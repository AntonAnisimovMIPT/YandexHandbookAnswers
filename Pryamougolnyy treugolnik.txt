#include <iostream>

int main()
{
    long long unsigned int a, b, c;
    std::cin >> a;
    std::cin >> b;
    std::cin >> c;
    if (((a*a + b*b == c*c) || (a*a + c*c == b*b) || (b*b + c*c == a*a)) && ((a*a + b*b + c*c != 0)))
    {
        std::cout << "YES";
    }
    else if ((a+b <= c) || (a+c <= b) || (c+b <= a))
    {
        std::cout << "UNDEFINED";
    }
    else
    {
        std::cout << "NO";
    }
}
